# Mobile Customer API Test Commands

# 1. Request OTP
curl -X POST http://localhost:8082/api/mobile/auth/request-otp \
  -H "Content-Type: application/json" \
  -d '{
    "mobileNumber": "9876543210",
    "deviceId": "test-device-123",
    "deviceType": "ANDROID",
    "appVersion": "1.0.0"
  }'

# 2. Verify OTP (replace OTP_CODE with actual OTP from logs)
curl -X POST http://localhost:8082/api/mobile/otp/verify \
  -H "Content-Type: application/json" \
  -d '{
    "mobileNumber": "9876543210",
    "otp": "OTP_CODE",
    "purpose": "LOGIN",
    "deviceId": "test-device-123"
  }'

# 3. Register Customer
curl -X POST http://localhost:8082/api/mobile/customer/register \
  -H "Content-Type: application/json" \
  -d '{
    "firstName": "John",
    "lastName": "Doe",
    "mobileNumber": "9876543210",
    "email": "john@example.com",
    "deviceId": "test-device-123",
    "appVersion": "1.0.0",
    "acceptTerms": true,
    "acceptPrivacy": true
  }'

# 4. Get App Configuration
curl -X GET http://localhost:8082/api/mobile/config

# 5. Login with Mobile (for existing customers)
curl -X POST http://localhost:8082/api/mobile/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "mobileNumber": "9876543210",
    "otp": "OTP_CODE",
    "deviceId": "test-device-123",
    "deviceType": "ANDROID",
    "rememberDevice": true
  }'
